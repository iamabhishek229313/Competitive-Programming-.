#include <iostream>
#include <string.h>
using namespace std ;
int main (){
    string the_code ; int k ;
    getline(cin,the_code);
    cin>>k;
    for(int i = 0 ;i < the_code.length() ;i++){
        //Taking the big alphabates .
        if(the_code[i]>=65 && the_code[i]<=90){
            if((the_code[i] + k) <= 90){
                the_code[i] = the_code[i] + k ;
            }
            else if ((the_code[i] + k) > 90){
                int tail = 90 - the_code[i] ;
                int copy_of_k = k ;
                 copy_of_k = k - tail - 1 ;
                the_code[i] = copy_of_k ;
            }
        }
        //Taking the small alphabates .
        
        else if (the_code[i]>=97 && the_code[i]<=122){
            if((the_code[i] + k) <= 122){
                the_code[i] = the_code[i] + k ;
            }
            else if ((the_code[i] + k) > 122){
                int tail = 122 - the_code[i] ;
                int copy_of_k = k ;
                 copy_of_k = k - tail - 1 ;
                the_code[i] = copy_of_k ;
            }
        }
        
        //Taking up the Numbers .
        else if (the_code[i]>=0 && the_code[i]<=9){
            if((the_code[i] + k) <= 9){
                the_code[i] = the_code[i] + k ;
            }
            else if ((the_code[i] + k) > 9){
                int tail = 9 - the_code[i] ;
                int copy_of_k = k ;
                 copy_of_k = k - tail - 1 ;
                the_code[i] = copy_of_k ;
            }
        }
        //Doing nothing for other things .
        else{
            
        }
    }
    cout<<the_code<<endl;
    
    return  0 ;
}
